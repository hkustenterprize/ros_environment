cmake_minimum_required(VERSION 2.8)
project(CV_AR)
set(CMAKE_BUILD_TYPE Debug)
SET(CMAKE_CXX_FLAGS "-std=c++0x -pthread")
find_package (Threads REQUIRED)
find_package(OpenCV REQUIRED)

aux_source_directory(./src DIR_SRCS)
include_directories(${PROJECT_SOURCE_DIR}/include)
FIND_LIBRARY(MVSDK ${PROJECT_SOURCE_DIR}/lib)

file(COPY ${PROJECT_SOURCE_DIR}/build/settings.xml DESTINATION ${PROJECT_BINARY_DIR})
file(COPY ${PROJECT_SOURCE_DIR}/build/CamConfigs DESTINATION ${PROJECT_BINARY_DIR})

add_executable(CV_AR ${DIR_SRCS})

TARGET_LINK_LIBRARIES(CV_AR ${MY_LIB})
target_link_libraries(CV_AR ${OpenCV_LIBS})
target_link_libraries(CV_AR MVSDK)
